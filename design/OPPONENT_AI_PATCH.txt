def generate_opponent_loadout(opponent_name, table_position, is_home=False):
    archetype = TEAM_ARCHETYPES.get(opponent_name, "Balanced")
    priority  = ARCHETYPES[archetype]["priority"]

    if table_position <= 3:
        card_count, allow_rare = 4, True
    elif table_position <= 7:
        card_count, allow_rare = 3, random.random() < 0.5
    else:
        card_count, allow_rare = 2, False

    if is_home:
        card_count += 1

    picked = {}
    for card_id in priority:
        if len(picked) >= card_count:
            break
        card = SKILL_CARDS.get(card_id)
        if not card:
            continue
        if card.get("rarity") == "Rare" and not allow_rare:
            continue
        picked[card_id] = copy.deepcopy(card)

    if len(picked) < card_count:
        fallback = [cid for cid in SKILL_CARDS if cid not in picked]
        random.shuffle(fallback)
        for cid in fallback:
            if len(picked) >= card_count:
                break
            card = SKILL_CARDS[cid]
            if card.get("rarity") != "Rare" or allow_rare:
                picked[cid] = copy.deepcopy(card)

    return picked
